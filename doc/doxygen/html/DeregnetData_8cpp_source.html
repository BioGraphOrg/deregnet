78 void DeregnetData::read_score(string* pathToTsv, bool take_abs) {
79  if (!graph) {
80  cerr << "Error: call AvgDeregnetData::read_graph() first." << endl;
81  exit(INPUT_ERROR);
82  }
83  if (!pathToTsv) {
84  cerr << "No score file specified." << endl;
85  exit(INPUT_ERROR);
86  }
87  map<string, std::set<double>> score_map;
88  try {
89  ifstream tsv;
90  tsv.open( *pathToTsv );
91  string key;
92  double value;
93  while ( !tsv.eof() ) {
94  tsv >> key >> value;
95  if (take_abs)
96  value = std::abs(value);
97  if (score_map.find(key) != score_map.end())
98  score_map[key].insert(value);
99  else
100  score_map[key] = { value };
101  }
102  score = new NodeMap<double>(*graph);
103  for (NodeIt v(*graph); v != INVALID; ++v) {
104  std::set<double> score_set = score_map[(*nodeid)[v]];
105  if (score_set.size() == 0)
106  (*score)[v] = 0.0;
107  else
108  (*score)[v] = std::accumulate(score_set.begin(), score_set.end(), 0.0) / score_set.size();
109 
110 // std::cout << (*nodeid)[v] << " : " << (*score)[v] << std::endl;
111  }
112  tsv.close();
113  }
114  catch ( ... ) {
115  cerr << "Could not read score file " << pathToTsv << "." << endl;
116  exit(INPUT_ERROR);
117  }
118 }

146 void DeregnetData::reverse_graph() {
147  map<Node,Node> node_correspondence;
148  revgraph = new Graph();
149  revnodeid = new NodeMap<string>(*revgraph);
150  for (NodeIt v(*graph); v != INVALID; ++v) {
151  Node revv = revgraph->addNode();
152  (*revnodeid)[revv] = (*nodeid)[v];
153  node_correspondence[v] = revv;
154  }
155  for (NodeIt v(*graph); v != INVALID; ++v) {
156  Node revv = node_correspondence[v];
157  for (InArcIt a(*graph, v); a != INVALID; ++a) {
158  Node s = graph->source(a);
159  Node revs = node_correspondence[s];
160  revgraph->addArc(revv,revs);
161  }
162  }
163  original_graph = graph;
164  original_nodeid = nodeid;
165  graph = revgraph;
166  nodeid = revnodeid;
167  swap(receptors, terminals);
168 }
