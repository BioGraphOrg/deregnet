70 void LazyConstraintCallback::callback() {
71  try {
72  if (where == GRB_CB_MIPSOL) { // new incumbent found
73  NodeFilter solution_filter(*graph);
74  selected_nodes = {};
75  get_solution_filter(solution_filter);
76  InducedSubgraph current_subgraph(*graph, solution_filter);
77  InducedSubgraph::NodeMap<int> component_map(current_subgraph);
78  int num_components { lemon::stronglyConnectedComponents(current_subgraph, component_map) };
79  if (num_components > 1)
80  check_and_set_lazy_constr(num_components, component_map);
81  }
82  else if (gap_cut && where == GRB_CB_MIP) {
83  double best_objective { getDoubleInfo(GRB_CB_MIP_OBJBST) };
84  double best_bound { getDoubleInfo(GRB_CB_MIP_OBJBND) };
85  if ( abs(best_objective - best_bound) < (*gap_cut) * (1.0 + abs(best_objective)) ) {
86  cout << "Achieved " << *gap_cut << " gap. Stopping optimization." << endl;
87  abort();
88  }
89  }
90  }
91  catch (GRBException e) {
92  std::cout << "Error number: " << e.getErrorCode() << std::endl;
93  std::cout << e.getMessage() << std::endl;
94  }
95  catch (...) {
96  std::cout << "Error during callback" << std::endl;
97  }
98 }

100 void LazyConstraintCallback::check_and_set_lazy_constr(const int num_components,
101  const InducedSubgraph::NodeMap<int>& component_map) {
102  for(int k = 0; k < num_components; k++) {
103  std::set<Node> component;
104  std::set<Node> parents;
105  std::set<Node> global_parents;
106  bool more_than_one_node = get_component_nodes(component_map, component, k); // violation of POLS
107  if (!is_root_component(component) && more_than_one_node) {
108  get_parents(component, parents, global_parents);
109  if (parents.size() == 0)
110  set_lazy_constraint(component, global_parents);
111  }
112  }
113 }

133 void LazyConstraintCallback::get_parents(const std::set<Node>& component,
134  std::set<Node>& parents,
135  std::set<Node>& global_parents) {
136  for (auto v : component) {
137  for (InArcIt a(*graph, v); a != INVALID; ++a) {
138  Node u = graph->source(a);
139  if (component.find(u) == component.end()) {
140  global_parents.insert(u);
141  if (find(selected_nodes.begin(), selected_nodes.end(), u) != selected_nodes.end())
142  parents.insert(u);
143  }
144  }
145  }

162 void LazyConstraintCallbackRoot::set_lazy_constraint(const std::set<Node>& component,
163  const std::set<Node>& global_parents) {
164  GRBLinExpr component_expr;
165  GRBLinExpr global_parents_expr;
166  for (auto v : component)
167  component_expr += (*x)[v];
168  for (auto v : global_parents)
169  global_parents_expr += (*x)[v];
170  GRBLinExpr lazy_constr_lhs = component_expr - global_parents_expr;
171  int comp_size = component.size();
172  addLazy(lazy_constr_lhs <= comp_size - 1);
173 }
