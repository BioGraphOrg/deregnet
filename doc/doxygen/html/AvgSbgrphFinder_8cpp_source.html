69 void AvgSbgrphFinder::Data::read_score(string* pathToTsv) {
70  if (!graph) {
71  cerr << "Error: call AvgSbgrphFinder::Data::read_graph() first." << endl;
72  exit(INPUT_ERROR);
73  }
74  if (!pathToTsv) {
75  cerr << "No score file specified." << endl;
76  exit(INPUT_ERROR);
77  }
78  map<string, double> score_map;
79  try {
80  ifstream tsv;
81  tsv.open( *pathToTsv );
82  string key;
83  double value;
84  while ( !tsv.eof() ) {
85  tsv >> key >> value;
86  score_map[key] = value;
87  }
88  score = new NodeMap<double>(*graph);
89  for (NodeIt v(*graph); v != INVALID; ++v)
90  (*score)[v] = score_map[(*nodeid)[v]];
91  tsv.close();
92  }
93  catch ( ... ) {
94  cerr << "Could not read score file " << pathToTsv << "." << endl;
95  exit(INPUT_ERROR);
96  }
97 }

131 vector<Sbgrph> AvgSbgrphFinder::run(grbfrc::Algorithm algorithm, bool start_heuristic, string model_sense) {
132  // set up model
133  AvgSbgrphModel model(data.graph, data.score, data.nodeid, data.root);
134  model.createVariables();
135  model.addBaseConstraints(data.min_size, data.max_size);
136  if (data.include)
137  model.addIncludeConstraints(data.include);
138  if (data.exclude)
139  model.addExcludeConstraints(data.exclude);
140  if (!receptor_as_root) {
141  if (data.receptors)
142  model.addReceptorConstraints(data.receptors);
143  if (data.terminals)
144  model.addTerminalConstraints(data.terminals);
145 
146  }
147  else if (receptor_as_root) {
148  model.reverseGraph();
149  if (data.terminals && !data.root)
150  model.addReceptorConstraints(data.terminals);
151  if (data.receptors)
152  model.addTerminalConstraints(data.receptors);
153  }
154 
155  vector<Sbgrph> subgraphs;
156  // find optimal subgraph (modulo time_limit and/or gap_cut)
157 
158  if ( model.solve(algorithm,
159  start_heuristic,
160  data.time_limit,
161  data.gap_cut,
162  model_sense) ) {
163  subgraphs.push_back ( toSbgrph( model.getCurrentSolution(), "optimal" ) );
164  }
165  else {
166  cerr << "No optimal subgraph could be found." << endl;
167  exit(NO_OPTIMAL_SUBGRAPH_FOUND);
168  }
169  // find suboptimal subgraphs
170  set<string> nodes_so_far { subgraphs.back().nodes }; // rethink this policy
171  for (int i = 0; i < data.num_subopt_iter; i++) {
172  model.addSuboptimalityConstraint(nodes_so_far, data.max_overlap);
173  if ( model.solve(algorithm,
174  start_heuristic,
175  data.time_limit,
176  data.gap_cut,
177  model_sense) ) {
178  AvgSbgrphModel::Solution current { model.getCurrentSolution() };
179  subgraphs.push_back(toSbgrph( current, "suboptimal_" + to_string(i) ));
180  for (auto node: current.nodes)
181  nodes_so_far.insert(node);
182  }
183  else {
184  cout << "With these parameters, no more than " << i
185  << " suboptimal subgraphs could be found." << endl;
186  break;
187  }
188  }
189  return subgraphs;
190 }

192 Sbgrph AvgSbgrphFinder::toSbgrph( AvgSbgrphModel::Solution solution, string signature ) {
193  Sbgrph subgraph;
194  subgraph.signature = signature;
195  subgraph.rootid = solution.rootid;
196  subgraph.nodes = solution.nodes;
197  subgraph.total_score = solution.total_score;
198  subgraph.avg_score = solution.avg_score;
199  if (data.receptors)
200  for (auto nodeid : subgraph.nodes) {
201  Node node;
202  if (getNodeById(data.graph, data.nodeid, nodeid, &node))
203  if (data.receptors->find(node) != data.receptors->end())
204  subgraph.receptors.insert(nodeid);
205  }
206  if (data.terminals)
207  for (auto nodeid : subgraph.nodes) {
208  Node node;
209  if (getNodeById(data.graph, data.nodeid, nodeid, &node))
210  if (data.terminals->find(node) != data.terminals->end())
211  subgraph.receptors.insert(nodeid);
212  }
213  for (auto edge : solution.edgelist) {
214  if (data.receptor_as_root)
215  subgraph.edgelist.insert(std::make_pair(edge.second, edge.first));
216  else
217  subgraph.edgelist = solution.edgelist;
218  }
219  return subgraph;
220 }
