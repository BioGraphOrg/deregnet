86 template <typename ModelType, typename Data>
87 void DeregnetFinder<ModelType, Data>::run_optimal_init(std::pair<Node, std::set<Node> >** start_solution) {
88  model->createVariables();
89  model->addBaseConstraints();
90  if (data->include)
91  model->addIncludeConstraints();
92  if (data->exclude)
93  model->addExcludeConstraints();
94  if (data->receptors && !data->root)
95  model->addReceptorConstraints();
96  if (data->terminals)
97  model->addTerminalConstraints();
98  if (data->start_heuristic)
99  find_start_solution(start_solution);
100  if (data->start_heuristic && !(*start_solution))
101  std::cout << "No heuristic start solution found.\n" << std::endl;
102 }

144 template <typename ModelType, typename Data>
145 std::vector<Subgraph> DeregnetFinder<ModelType, Data>::run() {
146  std::pair<Node, std::set<Node>>* start_solution { nullptr };
147  // find optimal subgraph
148  run_optimal_init(&start_solution);
149 
150  /*
151  std::cout << "# nodes in start: " << (start_solution->second).size() << std::endl;
152  write2sif(data->graph, start_solution->second, data->nodeid, "start.sif");
153  */
154 
155  std::vector<Subgraph> subgraphs;
156  bool solve_successful { model->solve(start_solution) };
157  run_optimal_windup(solve_successful, &subgraphs);
158  // find suboptimal subgraphs
159  std::set<std::string> nodes_so_far { subgraphs.back().nodes };
160  for (int i = 0; i < data->num_subopt_iter; ++i) {
161  start_solution = nullptr;
162  run_suboptimal_init(&start_solution, &nodes_so_far);
163  // solve for next suboptimal solution
164  solve_successful = model->solve(start_solution);
165  if (!run_suboptimal_windup(solve_successful, &subgraphs, &nodes_so_far, i))
166  break;
167  }
168  return subgraphs;
169 }

171 template <typename ModelType, typename Data>
172 Subgraph DeregnetFinder<ModelType, Data>::to_subgraph(Solution solution, std::string signature) {
173  Subgraph subgraph;
174  subgraph.signature = signature;
175  subgraph.rootid = solution.rootid;
176  subgraph.nodes = solution.nodes;
177  subgraph.total_score = solution.total_score;
178  subgraph.avg_score = solution.avg_score;
179  if (!data->receptor_as_root)
180  swap(data->receptors, data->terminals);
181  if (data->receptors)
182  for (auto nodeid : subgraph.nodes) {
183  Node node;
184  if (getNodeById(data->original_graph, data->nodeid, nodeid, &node))
185  if (data->receptors->find(node) != data->receptors->end())
186  subgraph.receptors.insert(nodeid);
187  }
188  if (data->terminals)
189  for (auto nodeid : subgraph.nodes) {
190  Node node;
191  if (getNodeById(data->original_graph, data->nodeid, nodeid, &node))
192  if (data->terminals->find(node) != data->terminals->end())
193  subgraph.terminals.insert(nodeid);
194  }
195  if (!data->receptor_as_root)
196  for (auto edge : solution.edgelist)
197  subgraph.edgelist.insert(std::make_pair(edge.second, edge.first));
198  else
199  subgraph.edgelist = solution.edgelist;
200  return subgraph;
201 }
