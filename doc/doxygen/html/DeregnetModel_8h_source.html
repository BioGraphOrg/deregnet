134 template <typename Model, typename Data>
135 GRBLinExpr DeregnetModel<Model, Data>::setBaseConstraintsRootCommon() {
136  GRBLinExpr subgraph_size_lhs;
137  for (NodeIt v(*graph); v != INVALID; ++v) {
138  subgraph_size_lhs += x[v];
139  GRBLinExpr parent_sum;
140  for (InArcIt a(*graph,v); a != INVALID; ++a)
141  parent_sum += x[graph->source(a)];
142  if (v != *root)
143  model.addConstr(x[v] - parent_sum <= 0);
144  }
145  model.addConstr(x[*root] == 1);
146  return subgraph_size_lhs;
147 }

149 template <typename Model, typename Data>
150 GRBLinExpr DeregnetModel<Model, Data>::setBaseConstraintsNoRootCommon() {
151  GRBLinExpr subgraph_size_lhs;
152  GRBLinExpr exactly_one_root_lhs;
153  for (NodeIt v(*graph); v != INVALID; ++v) {
154  model.addConstr((*y)[v] <= x[v]);
155  subgraph_size_lhs += x[v];
156  exactly_one_root_lhs += (*y)[v];
157  GRBLinExpr parent_sum;
158  for (InArcIt a(*graph,v); a != INVALID; ++a)
159  parent_sum += x[graph->source(a)];
160  model.addConstr(x[v] - (*y)[v] - parent_sum <= 0);
161  }
162  model.addConstr(exactly_one_root_lhs == 1);
163  return subgraph_size_lhs;
164 }

192 template <typename Model, typename Data>
193 void DeregnetModel<Model, Data>::setStartSolution(std::pair<Node, std::set<Node>>* start_solution) {
194  for (NodeIt v(*graph); v != INVALID; ++v) {
195  if ((start_solution->second).find(v) != (start_solution->second).end())
196  x[v].set(GRB_DoubleAttr_Start, 1.0);
197  else
198  x[v].set(GRB_DoubleAttr_Start, 0.0);
199  if (!root) {
200  if (v == start_solution->first)
201  (*y)[v].set(GRB_DoubleAttr_Start, 1.0);
202  else
203  (*y)[v].set(GRB_DoubleAttr_Start, 0.0);
204  }
205  }
206 }

233 template <typename Model, typename Data>
234 void DeregnetModel<Model, Data>::addTerminalConstraints() { // consider using complement of terminals ...
235  for (NodeIt v(*graph); v != INVALID; ++v) {
236  std::set<Node>* terminals { data->terminals };
237  if (terminals->find(v) == terminals->end()) {
238  GRBLinExpr out_neighbor_expr;
239  for (OutArcIt a(*graph, v); a != INVALID; ++a)
240  out_neighbor_expr += x[graph->target(a)];
241  model.addConstr(x[v] - out_neighbor_expr <= 0);
242  }
243  }
244  model.update();
245 }

247 template <typename Model, typename Data>
248 void DeregnetModel<Model, Data>::setup_solve(std::pair<Node, std::set<Node>>* start_solution) {
249  if (data->model_sense == "min")
250  model.set(GRB_IntAttr_ModelSense, GRB_MINIMIZE);
251  else
252  model.set(GRB_IntAttr_ModelSense, GRB_MAXIMIZE);
253 
254  model.set(GRB_IntParam_LazyConstraints, 1);
255 
256  if (data->time_limit)
257  model.set(GRB_DoubleParam_TimeLimit, *(data->time_limit));
258 
259  if (start_solution)
260  setStartSolution(start_solution);
261 
262  if (root)
263  setCallbackRoot();
264  else
265  setCallbackNoRoot();
266  model.update();

270 template <typename Model, typename Data>
271 Solution DeregnetModel<Model, Data>::getCurrentSolution() {
272  Solution solution;
273  std::set<Node> nodes;
274  string rootid;
275  _getCurrentSolution(&rootid, &nodes, &solution);
276  for (auto v : nodes) {
277  std::string source = (*nodeid)[v];
278  for (OutArcIt a(*graph, v); a != INVALID; ++a) {
279  std::string target { (*nodeid)[graph->target(a)] };
280  if (solution.nodes.find(target) != solution.nodes.end())
281  solution.edgelist.insert( std::make_pair(source, target));
282  }
283  }
284  solution.rootid = rootid;
285  solution.avg_score = solution.total_score / solution.nodes.size();
286  return solution;
287 }

381 void DeregnetModel<GRBModel, DrgntData>::_getCurrentSolution(std::string* rootid, std::set<Node>* nodes, Solution* solution) {
382  for (NodeIt v(*graph); v != INVALID; ++v) {
383  if (x[v].get(GRB_DoubleAttr_X) >= 0.98) {
384  nodes->insert(v);
385  (solution->nodes).insert((*nodeid)[v]);
386  if (!root) {
387  if ((*y)[v].get(GRB_DoubleAttr_X) >= 0.98)
388  *rootid = (*nodeid)[v];
389  }
390  else
391  *rootid = (*nodeid)[*root];
392  }
393  }
394  solution->total_score = model.get(GRB_DoubleAttr_ObjVal);
395 }
