88 void AvgSbgrphModel::addBaseConstraintsRoot(int min_size,
89  int max_size) {
90  GRBLinExpr subgraph_size_lhs;
91  for (NodeIt v(*graph); v != INVALID; ++v) {
92  subgraph_size_lhs += x[v];
93  GRBLinExpr parent_sum;
94  for (InArcIt a(*graph,v); a != INVALID; ++a)
95  parent_sum += x[graph->source(a)];
96  filp.addConstr(x[v] - parent_sum <= 0);
97  }
98  filp.addConstr(subgraph_size_lhs <= max_size);
99  filp.addConstr(subgraph_size_lhs >= min_size);
100  filp.update();
101 }

103 void AvgSbgrphModel::addBaseConstraintsNoRoot(int min_size,
104  int max_size) {
105  GRBLinExpr subgraph_size_lhs;
106  GRBLinExpr exactly_one_root_lhs;
107  for (NodeIt v(*graph); v != INVALID; ++v) {
108  filp.addConstr((*y)[v] <= x[v]);
109  subgraph_size_lhs += x[v];
110  exactly_one_root_lhs += (*y)[v];
111  GRBLinExpr parent_sum;
112  for (InArcIt a(*graph,v); a != INVALID; ++a)
113  parent_sum += x[graph->source(a)];
114  filp.addConstr(x[v] - (*y)[v] - parent_sum <= 0);
115  }
116  filp.addConstr(subgraph_size_lhs <= max_size);
117  filp.addConstr(subgraph_size_lhs >= min_size);
118  filp.addConstr(exactly_one_root_lhs == 1);
119  filp.update();
120 }
