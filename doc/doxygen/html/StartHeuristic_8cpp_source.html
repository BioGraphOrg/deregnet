66 bool StartHeuristic::search_further() {
67  size--;
68  if (size > 0 && found_node)
69  return true;
70  else if (size > 0 && !found_node)
71  return false;
72  else if (size == 0) {

80 bool StartHeuristic::feasible_node(Node* node) {
81  if (exclude) {
