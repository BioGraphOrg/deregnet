119 void parse_options(int argc, char* argv[], Options& options, Data& data) {
120 
121  static struct option long_options[] {
122  {"graph", 1, 0, 'g'},
123  {"terminals", 1, 0, 'T'},
124  {"terminals-file", 1, 0, 0},
125  {"receptors", 1, 0, 'R'},
126  {"receptors-file", 1, 0, 0},
127  {"score", 1, 0, 's'},
128  {"algorithm", 1, 0, 'a'},
129  {"root", 1, 0, 'r'},
130  {"flip-orientation", 0, 0, 'f'},
131  {"output-dir", 1, 0, 'o'},
132  {"min-size", 1, 0, 'l'},
133  {"max-size", 1, 0, 'u'},
134  {"no-max-size", 0, 0, 0},
135  {"time-limit", 1, 0, 't'},
136  {"gap-cut", 1, 0, 'c'},
137  {"exclude", 1, 0, 'e'},
138  {"include", 1, 0, 'i'},
139  {"exclude-file", 1, 0, 0},
140  {"include-file", 1, 0, 0},
141  {"suboptimal", 1, 0, 'b'},
142  {"max-overlap-percentage", 1, 0, 'p'},
143  {"no-start-heuristic", 0, 0, 0},
144  {"help", 0, 0, 'h'},
145  {"version", 0, 0, 'v'},
146  {"model-sense", 0, 0, 0},
147  {NULL, 0, NULL, 0}
148  };
149 
150  int opt;
151  int option_index = 0;
152  string option_signature = "hvg:s:a:l:u:r:R:T:i:e:b:p:t:c:o:";
153  while ( (opt = getopt_long(argc, argv, option_signature.c_str(),
154  long_options, &option_index)) != -1) {
155  switch (opt) {
156  case 0: {
157  string no_start_heuristic { "no-start-heuristic" };
158  string no_max_size { "no-max-size" };
159  string terminals_file { "terminals-file" };
160  string receptors_file { "receptors-file" };
161  string include_file { "include-file" };
162  string exclude_file { "exclude-file" };
163  string model_sense { "model-sense" };
164  // --no-start-heuristic
165  if ( strcmp(long_options[option_index].name, no_start_heuristic.c_str()) == 0 )
166  data.start_heuristic = false;
167  // --no-max-size
168  else if ( strcmp(long_options[option_index].name, no_max_size.c_str()) == 0)
169  options.no_max_size = true;
170  // --terminals-file
171  else if ( strcmp(long_options[option_index].name, terminals_file.c_str()) == 0)
172  register_node_set_file(&options.terminals, optarg);
173  // --receptors-file
174  else if ( strcmp(long_options[option_index].name, receptors_file.c_str()) == 0)
175  register_node_set_file(&options.receptors, optarg);
176  // --include-file
177  else if ( strcmp(long_options[option_index].name, include_file.c_str()) == 0)
178  register_node_set_file(&options.include, optarg);
179  // --exclude-file
180  else if ( strcmp(long_options[option_index].name, exclude_file.c_str()) == 0)
181  register_node_set_file(&options.exclude, optarg);
182  else if ( strcmp(long_options[option_index].name, model_sense.c_str()) == 0)
183  data.model_sense = string(optarg);
184  break;
185  }
186  case 'g':
187  // -g,--graph
188  options.lgf_file = new string(optarg);
189  break;
190  case 'T':
191  // -T,--terminals
192  register_node_set(&options.terminals, optarg);
193  break;
194  case 'R':
195  // -R,--receptors
196  register_node_set(&options.receptors, optarg);
197  break;
198  case 's':
199  // -s,--score
200  options.score_tsv = new string(optarg);
201  break;
202  case 'a': {
203  // -a,--algorithm
204  string gcc { "gcc" };
205  string dta { "dta" };
206  string ovt { "ovt" };
207  if ( strcmp(optarg, gcc.c_str()) == 0 )
208  data.algorithm = Algorithm::GCC;
209  else if ( strcmp(optarg, dta.c_str()) == 0 )
210  data.algorithm = Algorithm::DTA;
211  else if ( strcmp(optarg, ovt.c_str()) == 0 )
212  data.algorithm = Algorithm::OVT;
213  else {
214  cout << "\nUnknown algorithm option provided:" << endl;
215  cout << "-a,--algorithm < gcc | dta | ovt >" << endl;
216  print_help();
217  exit(OPTION_ERROR);
218  }
219  break;
220  }
221  case 'r':
222  // -r,--root
223  options.root_id = new string(optarg);
224  break;
225  case 'f':
226  // -f,--flip-orientation
227  options.receptor_as_root = false;
228  break;
229  case 'o':
230  // -o,--output-dir
231  options.outdir = new string(optarg);
232  break;
233  case 'l':
234  // -l,--min-size
235  data.min_size = atoi( optarg );
236  break;
237  case 'u':
238  // -u,--max-size
239  data.max_size = atoi( optarg );
240  break;
241  case 't':
242  // -t,--time-limit
243  data.time_limit = new double( atof( optarg ) );
244  break;
245  case 'c':
246  // -c,--gap-cut
247  data.gap_cut = new double( atof( optarg ) );
248  break;
249  case 'e':
250  // -e,--exclude
251  register_node_set(&options.exclude, optarg);
252  break;
253  case 'i':
254  // -i,--include
255  register_node_set(&options.include, optarg);
256  break;
257  case 'b':
258  // -b,--suboptimal
259  data.num_subopt_iter = atoi( optarg );
260  break;
261  case 'p':
262  // -p,--max-overlap-percentage
263  data.max_overlap = atof( optarg );
264  break;
265  case 'h':
266  // -h,--help
267  print_help();
268  exit(EXIT_SUCCESS);
269  case 'v':
270  // -v,--version
271  print_version();
272  exit(EXIT_SUCCESS);
273  case '?':
274  break;
275  }
276  }
277 }

281 void finalize_data(Options& options, Data& data) {
282  data.read_graph(options.lgf_file);
283  data.read_score(options.score_tsv);
284  data.get_node_set(&data.terminals, options.terminals);
285  data.get_node_set(&data.receptors, options.receptors);
286  data.get_node_set(&data.include, options.include);
287  data.get_node_set(&data.exclude, options.exclude);
288  data.get_root(options.root_id);
289  if (options.no_max_size)
290  data.max_size = countNodes(*data.graph);
291  if (data.algorithm == Algorithm::DTA)
292  data.gap_cut = nullptr;
293 }

297 void writeSubgraphs(vector<Subgraph>& subgraphs, string* outdirp) {
298  char cwd[2048];
299  string outdir;
300  if (getcwd(cwd, sizeof(cwd)) != NULL)
301  outdir = string(cwd);
302  if (outdirp)
303  outdir = *outdirp;
304  struct stat st = {0};
305  if (stat(outdir.c_str(), &st) == -1)
306  mkdir(outdir.c_str(), 0700);
307  string outdir_plain { outdir + "/plain" };
308  if (stat(outdir_plain.c_str(), &st) == -1)
309  mkdir(outdir_plain.c_str(), 0700);
310  for (auto subgraph : subgraphs) {
311  subgraph.writeToFile(outdir);
312  subgraph.writeToFile(outdir_plain, true);
313  }
314 }
