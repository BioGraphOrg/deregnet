91 bool AvgStartHeuristic::feasible_node(Node* node) {
92  double new_avg { update_avg(node) };
93  int current_size { static_cast<int>(start_solution->size()) };
94  if (current_size >= min_size && new_avg < current_avg)
95  return false;
96  current_avg = new_avg;
97  if (exclude) {
98  if (start_solution->find(*node) == start_solution->end() &&
99  exclude->find(*node) == exclude->end()) {
100  return true;
101  }
102  }
103  else
104  if (start_solution->find(*node) == start_solution->end())
105  return true;
106  return false;
107 }

109 double AvgStartHeuristic::update_avg(Node* node) {
110  int current_size { static_cast<int>(start_solution->size()) };
111  return ((*score)[*node] + current_size * current_avg) / (current_size + 1);
112 }
