63 Node* AvgSuboptimalStartHeuristic::get_best_root() {
64  double best;
65  Node* best_node { nullptr };
66  bool overlap_matters { max_overlap < 1.0 / nodes_so_far->size() };
67  if (receptors)
68  for (auto v : *receptors) {
69  if ( is_overlap_node(&v) && overlap_matters )
70  continue;
71  else
72  update_best_node(&best_node, &v, &best);
73  }
74  else
75  for (NodeIt v(*graph); v != INVALID; ++v) {
76  if ( is_overlap_node(&v) && overlap_matters )
77  continue;
78  else
79  update_best_node(&best_node, &v, &best);
80  }
81  return best_node;
82 }

107 bool AvgSuboptimalStartHeuristic::feasible_node(Node* node) {
108  double new_avg { update_avg(node) };
109  int current_size { static_cast<int>(start_solution->size()) };
110  if (new_avg < current_avg && current_size > min_size)
111  return false;
112  current_avg = new_avg;
113 
114  bool overlap_node { is_overlap_node(node) };
115  double potential_overlap { static_cast<double>( current_overlapping_nodes / start_solution->size() ) };
116  if (exclude) {
117  if (overlap_node) {
118  if ( start_solution->find(*node) == start_solution->end()
119  && potential_overlap < max_overlap
120  && exclude->find(*node) == exclude->end()) {
121  current_overlapping_nodes++;
122  return true;
123  }
124  }
125  else {

127  return true;

129  }
130  else {

133  && potential_overlap < max_overlap) {

135  return true;
136  }
