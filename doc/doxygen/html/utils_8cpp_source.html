49 set<string> split_string(const string& str,
50  const string& sep) {
51  set<string> split;
52  string::size_type start { 0 };
53  string::size_type last_sep { 0 };
54  while (last_sep != string::npos) {
55  last_sep = str.find_first_of(sep, start);
56  split.insert( str.substr(start, last_sep - start) );
57  start = last_sep + 1;
58  }
59  return split;
60 }
