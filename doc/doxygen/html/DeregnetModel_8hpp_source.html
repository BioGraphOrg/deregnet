234 template <typename Model, typename Data>
235 GRBLinExpr DeregnetModel<Model, Data>::setBaseConstraintsRootCommon() {
236  GRBLinExpr subgraph_size_lhs;
237  for (NodeIt v(*graph); v != INVALID; ++v) {
238  subgraph_size_lhs += x[v];
239  GRBLinExpr parent_sum;
240  for (InArcIt a(*graph,v); a != INVALID; ++a)
241  parent_sum += x[graph->source(a)];
242  if (v != *root)
243  model.addConstr(x[v] - parent_sum <= 0);
244  }
245  model.addConstr(x[*root] == 1);
246  return subgraph_size_lhs;
247 }

249 template <typename Model, typename Data>
250 GRBLinExpr DeregnetModel<Model, Data>::setBaseConstraintsNoRootCommon() {
251  GRBLinExpr subgraph_size_lhs;
252  GRBLinExpr exactly_one_root_lhs;
253  for (NodeIt v(*graph); v != INVALID; ++v) {
254  model.addConstr((*y)[v] <= x[v]);
255  subgraph_size_lhs += x[v];
256  exactly_one_root_lhs += (*y)[v];
257  GRBLinExpr parent_sum;
258  for (InArcIt a(*graph,v); a != INVALID; ++a)
259  parent_sum += x[graph->source(a)];
260  model.addConstr(x[v] - (*y)[v] - parent_sum <= 0);
261  }
262  model.addConstr(exactly_one_root_lhs == 1);
263  return subgraph_size_lhs;
264 }

337 void DeregnetModel<Model, Data>::addTerminalConstraints() { // consider using complement of terminals ...
338  for (NodeIt v(*graph); v != INVALID; ++v) {
339  std::set<Node>* terminals { data->terminals };
340  if (terminals->find(v) == terminals->end()) {
341  GRBLinExpr out_neighbor_expr;
342  for (OutArcIt a(*graph, v); a != INVALID; ++a)
343  out_neighbor_expr += x[graph->target(a)];
344  model.addConstr(x[v] - out_neighbor_expr <= 0);
345  }
346  }
347  model.update();

402 template <typename Model, typename Data>
403 Solution DeregnetModel<Model, Data>::getCurrentSolution() {
404  Solution solution;
405  std::set<Node> nodes;
406  std::string rootid;
407  _getCurrentSolution(&rootid, &nodes, &solution);
408  for (auto v : nodes) {
409  std::string source = (*nodeid)[v];
410  for (OutArcIt a(*graph, v); a != INVALID; ++a) {
411  std::string target { (*nodeid)[graph->target(a)] };
412  if (solution.nodes.find(target) != solution.nodes.end())
413  solution.edgelist.insert( std::make_pair(source, target));
414  }
415  }
416  solution.rootid = rootid;
417  solution.avg_score = solution.total_score / solution.nodes.size();
418  return solution;
419 }

537 template <> inline
538 bool DeregnetModel<grbfrc::FMILP, AvgdrgntData>::solve(std::pair<Node, std::set<Node>>* start_solution) {
539  setup_solve(start_solution);
540  // add callback here
541  grbfrc::GrbfrcCallback<Node>* cb;
542  if (root) {
543  std::vector<std::map<Node, GRBVar>>* vargrps = new std::vector<std::map<Node, GRBVar>>( { x } );
544  GrbfrcLazyConstraintCallbackRoot* callback = new GrbfrcLazyConstraintCallbackRoot(graph, root, data->gap_cut);
545  cb = new grbfrc::GrbfrcCallback<Node>(callback, vargrps);
546  }
547  else {
548  std::vector<std::map<Node, GRBVar>>* vargrps = new std::vector<std::map<Node, GRBVar>>( { x, *y } );
549  GrbfrcLazyConstraintCallbackNoRoot* callback = new GrbfrcLazyConstraintCallbackNoRoot(graph, data->gap_cut);
550  cb = new grbfrc::GrbfrcCallback<Node>(callback, vargrps);
551  }
552  double objlb = 0.0;
553  double objub = 0.0;
554  std::set<Node> maxnodes;
555  for (int k = 0; k <= data->min_size; ++k) {
556  double ub = -1000000.0;
557  for (NodeIt v(*graph); v != INVALID; ++v) {
558  if (maxnodes.find(v) == maxnodes.end() && score->operator[](v) > ub) {
559  ub = score->operator[](v);
560  objub += ub;
561  maxnodes.insert(v);
562  }
563  }
564  }
565 
566  objub = objub / data->min_size;
567  // std::cout << objub << std::endl;
568 
569  model.optimize(data->algorithm, cb, &objub, &objlb);
570  int status = model.get(GRB_IntAttr_Status);
571  return (status == GRB_OPTIMAL || status == GRB_INTERRUPTED || status == GRB_TIME_LIMIT);
572 }

576  for (NodeIt v(*graph); v != INVALID; ++v) {
