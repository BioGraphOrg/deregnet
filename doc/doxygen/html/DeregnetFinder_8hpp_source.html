172 template <typename ModelType, typename Data>
173 void DeregnetFinder<ModelType, Data>::run_optimal_init(std::pair<Node, std::set<Node> >** start_solution) {
174  model->createVariables(); // create binary variables corresponding to the nodes
175  model->addBaseConstraints(); // add base constraints to the model, ie connectivity etc.
176  // add optional constraints (ie. include, exclude, receptor and/or terminal constraints)
177  if (data->include)
178  model->addIncludeConstraints(); // which nodes to include in the subgraph, no matter what
179  if (data->exclude)
180  model->addExcludeConstraints(); // which nodes to exclude from the subgraph, no matter what
181  if (data->receptors && !data->root)
182  model->addReceptorConstraints(); // how to handle nodes which are 'receptors'
183  if (data->terminals)
184  model->addTerminalConstraints(); // how to handle nodes which are 'temrinals'
185  // optionally find heuristic start solution by greedy algorithm
186  if (data->start_heuristic)
187  find_start_solution(start_solution);
188  // in case a heuristic start solution was specfied but none was found, issue a warning.
189  // TODO: not necessarily always to std::cout ...
190  if (data->start_heuristic && !(*start_solution))
191  std::cout << "No heuristic start solution found.\n" << std::endl;
192 }

196 void DeregnetFinder<ModelType, Data>::run_optimal_windup(bool solve_successful, std::vector<Subgraph> *subgraphs) {
197  // if a solution was found update subgraphs data structure
198  if ( solve_successful ) {
199  subgraphs->push_back ( to_subgraph( model->getCurrentSolution(), "optimal" ) );
200  }
201  // if no solution was found, issue a message and end the program
202  // TODO: not necessarily always to std::cerr

204  else {
205  std::cerr << "No optimal subgraph could be found." << std::endl;
206  exit(NO_OPTIMAL_SUBGRAPH_FOUND);
207  }

212 void DeregnetFinder<ModelType, Data>::run_suboptimal_init(std::pair<Node, std::set<Node> > **start_solution,
213  std::set<std::string>* nodes_so_far) {
214  // make sure only a certain fraction of nodes contained in some already found subgraph
215  // will be contained this subgraph
216  model->addSuboptimalityConstraint(*nodes_so_far);
217  // optionally find heuristic start solution by greedy algorithm
218  if (data->start_heuristic)
219  find_suboptimal_start_solution(start_solution, nodes_so_far);
220  // in case a heuristic start solution was specfied but none was found, issue a warning.
221  // TODO: not necessarily always to std::cout ...
222  if (data->start_heuristic && !(*start_solution))
223  std::cout << "No heuristic start solution found.\n" << std::endl;

227 template <typename ModelType, typename Data>
228 bool DeregnetFinder<ModelType, Data>::run_suboptimal_windup(bool solve_successful,
229  std::vector<Subgraph>* subgraphs,
230  std::set<std::string>* nodes_so_far,
231  int i) {
232  // if a solution was found update subgraphs data structure
233  // Also, update the data structure tracking nodes contained in any already found subgraph
234  if ( solve_successful ) {
235  Solution current = model->getCurrentSolution() ;
236  subgraphs->push_back(to_subgraph( current, "suboptimal_" + std::to_string(i) ));
237  for (auto node: current.nodes)
238  nodes_so_far->insert(node);
239  return true;
240  }
241  // if no solution was found, issue a message
242  // TODO: not necessarily always to std::cout
243  else {
244  std::cout << "With these parameters, no more than " << i
245  << " suboptimal subgraphs could be found." << std::endl;
246  return false;
247  }

251 template <typename ModelType, typename Data>
252 std::vector<Subgraph> DeregnetFinder<ModelType, Data>::run() {
253  std::pair<Node, std::set<Node>>* start_solution { nullptr }; // potential heuristic start solution
254  // find optimal subgraph
255  run_optimal_init(&start_solution); // construct model, etc.
256 /*
257  std::ofstream file;
258  file.open("heuristic_start.nodes");
259  for (auto v : start_solution->second)
260  file << (*data->nodeid)[v] << "\n";
261  file.close();
262 */
263  std::vector<Subgraph> subgraphs; // data structure for found subgraphs
264  bool solve_successful { model->solve(start_solution) }; // try to find optimal solution
265  run_optimal_windup(solve_successful, &subgraphs); // handle result of optimal solution attempt
266  // find suboptimal subgraphs
267  std::set<std::string> nodes_so_far { subgraphs.back().nodes }; // tracking nodes contained in some already found subgraph
268  for (int i = 0; i < data->num_subopt_iter; ++i) {
269  start_solution = nullptr; // reset to NULL ... Where does the previous one go? Memory leak?
270  run_suboptimal_init(&start_solution, &nodes_so_far); // adapt model for finding next suboptimal, etc.
271  solve_successful = model->solve(start_solution); // try to find next suboptimal solution
272  if (!run_suboptimal_windup(solve_successful, &subgraphs, &nodes_so_far, i)) // once no suboptimal subgraph could be found, do not try again ;)
273  break;
274  }
275  return subgraphs;
276 }

279 template <typename ModelType, typename Data>
280 Subgraph DeregnetFinder<ModelType, Data>::to_subgraph(Solution solution, std::string signature) {
281  Subgraph subgraph;
282  subgraph.signature = signature;
283  subgraph.rootid = solution.rootid;
284  subgraph.nodes = solution.nodes;
285  subgraph.total_score = solution.total_score;
286  subgraph.avg_score = solution.avg_score;
287  if (!data->receptor_as_root)
288  swap(data->receptors, data->terminals);
289  if (data->receptors)
290  for (auto nodeid : subgraph.nodes) {
291  Node node;
292  if (getNodeById(data->original_graph, data->nodeid, nodeid, &node))
293  if (data->receptors->find(node) != data->receptors->end())
294  subgraph.receptors.insert(nodeid);
295  }
296  if (data->terminals)
297  for (auto nodeid : subgraph.nodes) {
298  Node node;
299  if (getNodeById(data->original_graph, data->nodeid, nodeid, &node))
300  if (data->terminals->find(node) != data->terminals->end())
301  subgraph.terminals.insert(nodeid);
302  }
303  if (!data->receptor_as_root)
304  for (auto edge : solution.edgelist)
305  subgraph.edgelist.insert(std::make_pair(edge.second, edge.first));
306  else
307  subgraph.edgelist = solution.edgelist;
308  return subgraph;
309 }

318 void DeregnetFinder<GRBModel, DrgntData>::find_start_solution(std::pair<Node,
319  std::set<Node>>** start_solution) {
320  std::function<bool(double, double)> cmp;
321  // Adjust comparison function based on whether to minimize or maximize
322  if (data->model_sense == "min") cmp = std::greater<double>();
323  else cmp = std::less<double>();
324  // Try to find start solution and when successful register it
325  StartHeuristic heuristic(data->graph,
326  data->score,
327  data->root,
328  data->exclude,
329  data->receptors,
330  cmp,
331  data->size);
332  if (heuristic.run())
333  *start_solution = heuristic.getStartSolution();
334 }
